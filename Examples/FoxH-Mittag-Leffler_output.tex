\begin{align*}
  a^*    & = 2-\rho \\

  \delta & = \text{ComplexInfinity} \\

  a_1^*  & = 1 \\
  a_2^*  & = 1-\rho \\
  \xi    & = \mu -1 \\
  c^*    & = \frac{1}{2} \\
\end{align*}
